Nu spottar koden ut i en sorterad lista.

Du använder fortfarande på samma sätt, dvs.
countPosts("https://www.rollspel.nu/threads/carmilla-vs-dracula-runda-1.84792/");

Code: 
async function countPosts(threadUrl, maxPages = 10) {
    let userPostCounts = {};
    let page = 1;
    
    while (page <= maxPages) {
        let url = page === 1 ? threadUrl : `${threadUrl}page-${page}`;
        console.log(`Fetching: ${url}`);
        
        try {
            let response = await fetch(url);
            if (!response.ok) throw new Error(`Failed to fetch page ${page}`);
            let html = await response.text();
            let parser = new DOMParser();
            let doc = parser.parseFromString(html, 'text/html');
            
            let posts = doc.querySelectorAll("article[data-author]");
            posts.forEach(post => {
                let username = post.getAttribute("data-author");
                userPostCounts[username] = (userPostCounts[username] || 0) + 1;
            });
            
            // Check if there is a next page
            let nextPageLink = doc.querySelector(".pageNav-jump--next");
            if (!nextPageLink) break;
        } catch (error) {
            console.error(error);
            break;
        }
        
        page++;
    }
    
    // Sort the results by post count in descending order
    let sortedCounts = Object.entries(userPostCounts)
        .sort((a, b) => b[1] - a[1]);
    
    console.log("Post count statistics:");
    sortedCounts.forEach(([user, count]) => console.log(`${user}: ${count}`));
    
    return sortedCounts;
}
